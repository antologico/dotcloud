117 		/**
118 			Carga de los ítems utilizando el servicio
119 			@function listar 
120 		    @memberof Selector
121 		*/
122 		this.listar = function ()
123 		{
124 			// Texto de la página
125 			// ------------------------------------------------------------------
126 			$("#texto_inicio").text(_("Seleccione el "+configurado+" que desea editar"));
127 
128 			$("#cargador_items").show ();
129 			$("#listado_ul_items").html ("");
130 
131 			$.post("servicios/"+configurado+".listar.php", null, 
132 				function(valor)
133 				{
134 					if (valor != "")
135 					{
136 						var items = eval(valor);
137 
138 						// Creamos un botón por cada ítem
139 						// y le asignamos una acción
140 
141 						for (i=0; i<items.length; i++)
142 						{
143 													var borrar 	= "";
144 													var clase  	= ""
145 													if ((items[i].rango == "Administrador") || (items[i].rango == "Editor"))
146 													{
147 														borrar 	= "<input type='button' id='borrar_"+items[i].id+"' rel='"+items[i].id+"' class='boton_borrar' />";
148 														clase	= "boton_entrar_borrar";
149 													}
150 													
151 
152 
153 													$("#listado_ul_items").append("<li><input type='button' class='boton_entrar "+clase+"' id='item_"+items[i].id+"' name='item_"+items[i].id+"' rel='"+items[i].id+"' value='"+items[i].nombre+"' />"+borrar+"</li>");											
154 													// Acción del botón: ir al editor
155 
156 													$('#item_'+items[i].id).click (function() 
157 													{ 
158 														destino = $(this).attr('rel');
159 														autoref.editarItem (destino);
160 													});
161 
162 
163 													if (borrar != "")
164 													$('#borrar_'+items[i].id).click (function() 
165 													{ 
166 														var id = $(this).attr('rel');
167 														AttentionBox.showMessage( _("¿Está seguro de borrarlo?"),
168 														{
169 														    buttons : 
170 														    [
171 														        { caption : _("Cancelar"), cancel: true },
172 														        { caption : _("Borrar") },
173 														    ],
174 															callback: function(action, inputs)
175 													    	{
176 
177 													    		if (action == _("Borrar"))
178 													    		{
179 																	// Solicitud de borrado
180 																	$("#cargador_items").show ();
181 																	$("#listado_ul_items").html("");
182 																	
183 
184 																	$.post("servicios/"+configurado+".gestionar.php", 
185 																		{
186 																			id: id,
187 																			accion: "borrar"
188 																		}, 
189 																		function(valor)
190 																		{
191 																			$("#cargador_items").hide ();
192 																			if (valor+0 != 0)
193 																				AttentionBox.showMessage( _("No se pudo borrar")+". Error:"+valor);
194 																			
195 																			autoref.listar();
196 																		});
197 																}
198 															}
199 														});
200 													});
201 													
202 						}
203 					}
204 
205 					$("#cargador_items").hide ();
206 			});
207 
208 			$("#form_nuevo_nuevo").click (function ()
209 			{
210 				autoref.asignarAccionNuevo ();
211 			});
212 		}

247 			@function editarItem 
248 		    @param {string} [destino] Página de destino para la edición (sólo para proyectos)

263 					break;								
264 			}
265 		}

271 		/**
272 			Controla la creación de un nuevo proyecto
273 			@function nuevoProyecto
274 		    @memberof Selector
275 		*/
276 		this.nuevoProyecto = function ()
277 		{	
278 				
279 					cargador.activar ();
280 
281 					$.post("servicios/compilador.listar.php", null, 
282 						function(compiladores)
283 						{
284 							var compiladores = eval (compiladores);
285 							
286 							cargador.desactivar ();
287 
288 							if (compiladores+0 != "-1")
289 							{	
290 								// Creamos la lista de compiladores
291 								var miscompiladores = {};
292 
293 								for (i=0; i<compiladores.length; i++)
294 								{
295 									miscompiladores[compiladores[i].nombre] = compiladores[i].id;
296 								}
297 
298 								AttentionBox.showMessage( _("Crear un nuevo proyecto"),
299 								{
300 								    inputs : 
301 								    [
302 								    	{ caption : _("Nombre"), id: "nombre_archivo_nuevo" },
303 								    	{ caption : _("Compilador"), rel:"proyecto.idcompilador", id: "conf_proyecto_idcompilador", 
304 								        			type: "select", values: miscompiladores },
305 								    ],
306 								    buttons : 
307 								    [
308 								        { caption : _("Cancelar"), cancel: true },
309 								        { caption : _("Crear proyecto") },
310 								    ],
311 									callback: function(action, inputs)
312 		    						{
313 		    							if (action == _("Crear proyecto"))
314 		    								if ((inputs[0].value != "") && (inputs[1].value != ""))
315 											{	
316 												$.post("servicios/proyecto.gestionar.php", 
317 												{
318 													nombre: inputs[0].value,
319 													idcompilador: inputs[1].value,
320 													accion: "crear"
321 												}, 
322 												function(valor)
323 												{
324 													if (valor+0 != 0)
325 														AttentionBox.showMessage( _("No se pudo crear el proyecto")+". Error:"+valor);
326 													else
327 														autoref.listar();
328 												});
329 											}
330 										
331 									}
332 								});
333 							}
334 							else 
335 								AttentionBox.showMessage( _("No se pudieron optener los compiladores")+". Error:"+compiladores);
336 					});
337 
338 		}
