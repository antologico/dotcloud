1 /**
  2  	Controlador del GUI
  3 	
  4 	@author Antonio Juan Sánchez Martín
  5 	@class Cargador
  6 	@version 0.9
  7 */
  8 
  9 function Controlador ()
 10 {
 11 		// Propiedades generales
 12 		// --------------------------------------------------------
 13 		var cargador 	= new Cargador ("cargador_controlador");	// Cargador para esperas, lo redefine
 14 		var autoref		= this;										// Autoreferencia del objeto, útil para llamadas dentro de funciones
 15 		var ayudaactiva = "inactiva";								// Activación de la ayuda en el hover
 16 		
 17 		/**
 18 			Carga el controlador, una vez se haya especificado el idioma. Es invocado desde this.mostrar()
 19 		    @function mostrar
 20 		    @param {string} idioma Idioma con el que se carga la GUI
 21 		    @memberof Controlador
 22 		*/
 23 		this.mostrar  = function (idioma)
 24 		{
 25 			// Reescribir por los controladores hijos
 26 		}
 27 
 28 		/**
 29 			Inicia los componenetes GUI
 30 		    @function iniciar
 31 		    @param {string} idioma Idioma con el que se carga la GUI
 32 		    @memberof Controlador
 33 		*/
 34 		this.iniciar = function ()
 35 		{	
 36 			$.get ('servicios/idioma.php', null, this.mostrar);
 37 		}
 38 
 39 
 40 		// --------------------------------------------------------
 41 		// Funciones de la ayuda
 42 		// --------------------------------------------------------
 43 
 44 		/**
 45 			Activa / desactiva las funciones de ayuda
 46 		    @function cambiarEstadoAyuda
 47 		    @memberof Controlador
 48 		*/
 49 		this.cambiarEstadoAyuda = function ()
 50 		{
 51 			this.mostrarAyuda ();
 52 
 53 			if (ayudaactiva == "inactiva") ayudaactiva = "activa";
 54 			else ayudaactiva = "inactiva";
 55 			
 56 			return ayudaactiva;
 57 
 58 		}
 59 
 60 		/**
 61 			Asigna a cada componente los eventos necesarios para mostrar la ayuda por pantalla
 62 		    @function mostrarAyuda
 63 		    @memberof Controlador
 64 		*/
 65 		this.mostrarAyuda = function ()
 66 		{
 67 			$(".ayuda").hover(function(){
 68 					if (ayudaactiva == "activa")
 69 					{
 70 						if ($(this).attr("ayuda"))
 71 						{
 72 							var elemento = this;
 73 							var pos = $(elemento).position();
 74 							// offset de los  contenedored
 75 							var width = $(elemento).outerWidth();
 76 							$("body").append ("<div id='ventanaayuda'></div>");
 77 							$("#ventanaayuda").addClass ("ventanaayuda");
 78 							$("#ventanaayuda").html ($(elemento).attr("ayuda"));
 79 							$("#ventanaayuda").fadeIn("slow");
 80 						}
 81 					}
 82 
 83 				}, 
 84 				function()
 85 				{
 86 					if ($("#ventanaayuda"))
 87 						$("#ventanaayuda").remove();	
 88 				});
 89 			
 90 		}
 91 
 92 		// --------------------------------------------------------
 93 		// Control de eventos
 94 		// --------------------------------------------------------
 95 
 96 		/**
 97 			Prepara la salida hacia otra página
 98 		    @function salir
 99 		    @memberof Controlador
100 		*/
101 		this.salir = function ()
102 		{
103 			cargador.activar ();
104 			$.post("servicios/sesion.cerrar.php", { }, function(valor)
105 							{
106 								cargador.desactivar();
107 								// Cuando se ha cerrado la session recargamos la página
108 								autoref.irA('index.php');
109 							}
110 					);
111 			
112 		}
113 
114 		/**
115 			Función de control y manejo de eventos del botón de idioma
116 			@function seleccionarIdioma
117 		    @memberof Controlador
118 		*/
119 		this.seleccionarIdioma = function ()
120 		{
121 			var misidiomas  = {};
122 			
123 			$.each(i18n, function (elidioma, elvalor)
124 			{
125 				misidiomas  [elidioma] = elidioma;
126 			});
127 
128 			AttentionBox.showMessage( _("Propiedades del idioma"),
129 			{
130 						    inputs : 
131 						    [
132 						    	{ caption : _("Idioma"), rel:"proyecto.idioma", id: "conf_gui_idioma", type: "select", values: misidiomas, selectvalue: idioma }
133 						    ],
134 						    buttons : 
135 						    [
136 						        { caption : _("Cancelar"), cancel: true },
137 						        { caption : _("Cambiar") },
138 						    ],
139 							callback: function(action, inputs)
140     						{
141     							if (action == _("Cambiar"))
142     							{
143     								// Enviamos el idioma nuevo al servicio cambiamos el valor definido para la GUI
144     								idioma = inputs[0].value;
145     								$.post("servicios/idioma.php", { idioma: idioma }, function(valor){ 
146     										if (valor == "") AttentionBox.showMessage( _("Error guardando el idioma de la session"));
147     										else AttentionBox.showMessage( _("La página se va a recargar"), {
148 										    buttons : 
149 										    [
150 										        { caption : _("Esperar"), cancel: true },
151 										        { caption : _("Recargar") },
152 										    ],
153 											callback: function(action)
154 				    						{
155 				    							if (action == _("Recargar"))
156     												autoref.recargar();
157     										}});
158 
159     								});
160     							}	
161     						}
162     		});
163 		}
164 
165 		/**
166 			Recarga la página con suavizado
167 			@function recargar
168 		    @memberof Controlador
169 		*/
170 		this.recargar = function ()
171 		{	
172 			$("body").fadeOut(1000, function () 
173 			{
174 				// Fade suave
175 			  	window.location.reload();
176 			});
177 		}
178 
179 		/**
180 			Carga una nueva página con suavizado
181 			@function irA
182 		    @param {string} pagina URL de la nueva página
183 		    @memberof Controlador
184 		*/
185 		this.irA = function (pagina)
186 		{
187 			$("body").fadeOut(1000, function () 
188 			{
189 				// Fade suave
190 			  	window.location = pagina;
191 			});
192 		}
193 }
