1 /**
  2 	Componente gráfico con la barra de herramientas de la GUI
  3 
  4 	@author Antonio Juan Sánchez Martín
  5 	@class BarraHerramientas
  6 	@version 0.9
  7 	@param {Controlador} contenedor Referencia al componente padre
  8 */
  9 	
 10 
 11 function BarraHerramientas (controlador)
 12 { 
 13 		// --------------------------------------------------------
 14 		// Creación de la ventana en el body
 15 		// $id de la ventana
 16 		$("body").append ('<div id="barra_herramientas_capa"><div id="logo"><img src="imagenes/logo_barra_herramientas.png" alt=".cloud" /></div><div id="iconos_rapidos"></div></div>');	
 17 		// --------------------------------------------------------
 18 		var controlador			= controlador;
 19 		var iconos_rapidos 		= $('#iconos_rapidos');
 20 		
 21 		/**
 22 			Mostrar capa de carga
 23 		    @function cargarIconosRapidos
 24 		    @param {string[]} proyectos Listado de los proyectos del sistema ["id", "nombre"]
 25 		    @memberof BarraHerramientas
 26 		*/
 27 		this.cargarIconosRapidos = function (proyectos)
 28 		{
 29 				
 30 				iconos_rapidos.append ('<select id="selector_proyectos" class="listado_proyectos"/></select>');
 31 				iconos_rapidos.append ('<input type="button" id="boton_usuarios" 	/>');
 32 				iconos_rapidos.append ('<input type="button" id="boton_crear" 		/>');
 33 				iconos_rapidos.append ('<input type="button" id="boton_guardar" 	/>');
 34 				iconos_rapidos.append ('<input type="button" id="boton_descargar" 	/>');
 35 				iconos_rapidos.append ('<input type="button" id="boton_eliminar" 	/>');
 36 				iconos_rapidos.append ('<input type="button" id="boton_compilar" 	/>');
 37 				iconos_rapidos.append ('<input type="button" id="boton_ejecutar" 	/>');
 38 				iconos_rapidos.append ('<input type="button" id="boton_configurar" 	/>');
 39 				iconos_rapidos.append ('<input type="button" id="boton_idioma" 		/>');
 40 				iconos_rapidos.append ('<input type="button" id="boton_ayuda" 		/>');
 41 				iconos_rapidos.append ('<input type="button" id="boton_salir" 		/>');
 42 
 43 				
 44 				
 45 				/// Selector
 46 				// -----------------------------------
 47 				// Creamos un botón por cada proyecto
 48 				// y le asignamos una acción
 49 				$("#selector_proyectos").append("<option>"+_("Seleccione proyecto")+"</option>");
 50 
 51 				for (i=0; i<proyectos.length; i++)
 52 				{
 53 					$("#selector_proyectos").append("<option value='"+proyectos[i].id+"'>"+proyectos[i].nombre+"</option>");
 54 				}
 55 				$("#selector_proyectos").change(function () {
 56 					if ($(this).val() != '')
 57 						controlador.cargarProyecto ($(this).val());
 58 				});
 59 				$('#salir').attr ("ayuda", _("Atención: Al cambiar de proyecto perderá todos los cambios de los archivos que no haya guardado"));
 60 				
 61 				// Botones y ayuda de éstos
 62 				// -----------------------------------
 63 				$("#boton_ayuda").click (function () 
 64 				{
 65 					$(this).removeClass ("ayuda_inactiva");
 66 					$(this).removeClass ("ayuda_activa");
 67 					estado = controlador.cambiarEstadoAyuda (); 
 68 					$(this).addClass ("ayuda_"+estado);
 69 				});
 70 				$('#boton_ayuda').attr ("ayuda", _("Mostrar/ocultar la información de ayuda"));
 71 				$('#boton_ayuda').addClass ("ayuda_inactiva");
 72 
 73 				$('#boton_crear').click (function()		{ controlador.crearArchivo(); } );
 74 				$('#boton_crear').attr ("ayuda", _("Crea un nuevo archivo o directorio en el proyecto"));
 75 
 76 				$('#boton_usuarios').click (function()	{ controlador.configurarUsuarios(); } );
 77 				$('#boton_usuarios').attr ("ayuda", _("Control y propiedades del acceso de usuarios al proyecto"));
 78 
 79 				$('#boton_guardar').click (function()		{ controlador.guardarArchivo(); } );
 80 				$('#boton_guardar').attr ("ayuda", _("Guardar el archivo que se está editando (Ctrl+S)"));
 81 
 82 				$('#boton_descargar').click (function() 	{ controlador.descargarArchivo(); } );
 83 				$('#boton_descargar').attr ("ayuda", _("Descargar el archivo en uso"));
 84 
 85 				$('#boton_eliminar').click (function()	{ controlador.eliminarArchivo(); } );
 86 				$('#boton_eliminar').attr ("ayuda", _("Eliminar el archivo en uso"));
 87 
 88 				$('#boton_compilar').click (function() 	{ controlador.ejecutar('compilacion'); } );	
 89 				$('#boton_compilar').attr ("ayuda", _("Compilar el proyecto (Ctrl+T)"));
 90 
 91 				$('#boton_ejecutar').click (function()	{ controlador.ejecutar('ejecucion'); } );
 92 				$('#boton_ejecutar').attr ("ayuda", _("Ejecutar el proyecto (Ctrl+R)"));
 93 
 94 				$('#boton_configurar').click (function() 	{ controlador.configurar(); } );	
 95 				$('#boton_configurar').attr ("ayuda", _("Configurar las opciones de compilación del proyecto"));
 96 
 97 				$('#boton_idioma').click (function() 		{ controlador.seleccionarIdioma(); } );
 98 				$('#boton_idioma').attr ("ayuda", _("Cambiar de idioma"));
 99 	
100 				$('#boton_salir').click (function() 		{ controlador.salir(); } );	
101 				$('#boton_salir').attr ("ayuda", _("Salir de la aplicación  (Ctrl+Q)"));
102 
103 
104 				$("#barra_herramientas_capa *").addClass("ayuda icono");	
105 
106 		}
107 
108 
109 		// --------------------------------------------------------
110 		
111 
112 }
