 92 		/**
 93 			Abre el editor con el archivo indicado
 94 		    @function cargarArchivo
 95 		    @param {string} nombre_archivo Archivo a cargar
 96 		    @param {int} [linea] Número de línea que aparecerá señalada
 97 		    @memberof EntornoTrabajo 
 98 		*/
 99 		this.cargarArchivo = function (nombre_archivo, linea)
100 		{
101 		
102 			// Comprobamos que el archivo no se haya abierto
103 			if ((lista_archivos_abiertos == undefined) || (lista_archivos_abiertos[nombre_archivo] == undefined))
104 			{
105 				// Carga el texto contenido en un archivo en el editor indicado
106 				cargador.activar();
107 
108 				// --------------------------------------------------------
109 				// Si no está creada la lista de archivos abiertos, la creamos
110 				if 	(this.lista_archivos_abiertos == undefined) 
111 					this.lista_archivos_abiertos = new Object();
112 				
113 				lista_archivos_abiertos[nombre_archivo] = 'abierto';
114 				
115 				// Debe ser asíncrono porque si no va mal la escritura en el editor
116 				$.post("servicios/archivo.leer.php", { archivo: nombre_archivo }, function(resultado)
117 					{
118 						if (resultado)
119 						{
120 							var resultado = eval(resultado);
121 							
122 							for(var json in resultado)
123 							{
124 									if (resultado[json]["errores"] != "0")
125 									{
126 										AttentionBox.showMessage(_("Error abriendo archivo.")+". Error: "+resultado[json]["errores"] );
127 										autoref.cerrarArchivo (nombre_archivo);
128 									}
129 									else
130 									{
131 										editor.crearEditor (nombre_archivo, resultado[json]["contenido"], linea); 
132 		
133 			            			}
134 			            			cargador.desactivar();
135 			            	}
136 			            }
137 					}
138 				);
139 			}
140 			else if (lista_archivos_abiertos[nombre_archivo] != undefined)
141 			{
142 				editor.activarEditor (nombre_archivo, linea); 
143 			}
144 			
145 		}

147 		/**
148 			Guarda el contenido de un archivo
149 		    @function surbirArchivo
150 		    @param {string} destino Archivo destino
151 		    @param {Obejct} manejadorArchivo Manejador del archivo (Ajax)
152 		    @memberof EntornoTrabajo 
153 		*/
154 		this.surbirArchivo = function (destino, manejadorArchivo)
155 		{
156 			if ((destino != undefined) && (manejadorArchivo))
157 			{
158 				var formData = new FormData();
159 				var xmlHttpRequest = new XMLHttpRequest();
160 				xmlHttpRequest.onreadystatechange = function (e) 
161 				{
162 					if (xmlHttpRequest.readyState == 4)
163 					{
164 						var error = xmlHttpRequest.responseText;
165 						if (error+0 != 0)
166 						AttentionBox.showMessage(_("Error abriendo archivo")+". Error: "+error );
167 						else
168 						explorador.cargarArchivosExplorador();
169 					}
170 				};
171 				xmlHttpRequest.open("POST", "servicios/archivo.cargar.php", true);
172                 formData.append("archivo", manejadorArchivo );
173                 formData.append("destino", destino );
174                 xmlHttpRequest.send(formData);
175 			}
176 			return false;
177 		}

184 		    @param {Obejct} contenido_archivo Contenido del archivo
185 		    @memberof EntornoTrabajo

203 				{

258 		/**
259 			Elimina un archivo 
260 		    @function eliminarArchivo
261 		    @param {string} nombre_archivo Archivo destino
262 		   	@memberof EntornoTrabajo 
263 		*/
264 		this.eliminarArchivo = function (nombre_archivo) 
265 		{
266 			if (nombre_archivo == null)
267 			{
268 				// Buscamos el archivo activo
269 				nombre_archivo  = editor.archivoActivo ();
270 			}
271 
272 			if (nombre_archivo != null)
273 			AttentionBox.showMessage( "¿"+_("Esta seguro de borrar ")+" "+nombre_archivo +"?",
274 			{
275 							buttons : 
276 						    [
277 						        { caption : _("Cancelar"), cancel: true },
278 						        { caption : _("Borrar") },
279 						    ],
280 							callback: function(action, inputs)
281     						{
282     							if (action ==  _("Cancelar"))
283     							{
284     								// No se hace nada, se cierra el diálogo y listo
285     							}
286     							else if (action ==  _("Borrar"))
287     							{
288     								// Cerramos el archivo
289 									if (autoref.lista_archivos_abiertos != undefined)
290 									if (autoref.lista_archivos_abiertos[nombre_archivo ] != undefined)
291 									if (autoref.lista_archivos_abiertos[nombre_archivo ] != null)
292 									{
293 										// Si el archivo está abierto, lo cerramos
294 										autoref.cerrarArchivo (nombre_archivo );
295 									}
296 
297 									// Si el elemento a borrar es un directorio, buscamos todos los archivos abiertos de ese directorio y los cerramos
298 									if (nombre_archivo.charAt(nombre_archivo.length-1)=="/")
299 									{
300 										if (autoref.lista_archivos_abiertos != undefined)
301 										$.each( autoref.lista_archivos_abiertos, function( fichero, valor) 
302 										{
303 											if (valor != null)
304 											if (fichero.indexOf(nombre_archivo ) == 0)
305 											{
306 												if(autoref.lista_archivos_abiertos)
307 												{
308 													autoref.cerrarArchivo (autoref.lista_archivos_abiertos[fichero]);
309 													autoref.cerrarSolapa (autoref.lista_archivos_abiertos[fichero]);
310 												}
311 											}
312 										});
313 										
314 									}
315 
316 									// Ponemos cargando el explorador
317 									explorador.explorar (false);
318 
319 									// Solicitamos la eliminacion
320 									$.post("servicios/archivo.eliminar.php", { archivo: nombre_archivo  }, function(error)
321 									{
322 										// En el retorno recargamos el explorador
323 										if (error+0 == 0)
324 										{
325 											AttentionBox.showMessage( _("Archivo eliminado"));
326 											if(autoref.lista_archivos_abiertos)
327 												autoref.cerrarArchivo (autoref.lista_archivos_abiertos[nombre_archivo]);
328 										}
329 										else
330 											AttentionBox.showMessage(_("No se ha podido borrar el archivo")+". Error: "+error );
331 										
332 										explorador.explorar (true);
333 										explorador.cargarArchivosExplorador();			
334 									});
335     							}
336     						}
337     		});
338 			else AttentionBox.showMessage(_("No hay documentos abiertos"));
339   
340 		}

436 		/**
437 			Lanza el servicio de ejecución remota
438 		    @function iniciarServicioEjecucion 
439 		    @param {string} tipo Tipo de acción
440 		    @memberof EntornoTrabajo 
441 		*/
442 		this.iniciarServicioEjecucion = function (tipo)
443 		{
444 			var servicio = null;
445 
446 			if ((tipo == 'ejecucion') || (tipo == 'compilacion')) servicio = tipo;
447 			
448 			// Lanzamos el servicio de compilación constantemente, hasta que se acabe
449 			// o hasta que se pare la compilación
450 			
451 			if (servicio != null)
452 			{
453 				if (ejecutando == 'continuar')
454 				{
455 					$.post("servicios/proyecto.ejecutar.php", { tipo : servicio }, function (resultado) 
456 					{
457 						resultado = eval (resultado);
458 
459 						if (resultado[0]["error"]+0 == 0)
460 						{
461 							resultado = eval (resultado);
462 							
463 							if (tipo == 'compilacion') ventana_ejecucion.listar (resultado[0]["datos"]);
464 							else ventana_ejecucion.imprimir (resultado[0]["datos"]);
465 
466 							if (resultado[0]["fin"] == 'true') ejecutando = 'parada';
467 
468 
469 							autoref.iniciarServicioEjecucion (servicio);
470 						}
471 						else 
472 						{
473 							AttentionBox.showMessage( _("Error en el servicio")+". Error: "+resultado[0]["error"]);
474 							ejecutando = 'parada';
475 							autoref.iniciarServicioEjecucion (servicio);
476 						}
477 					});
478 				}
479 				else if (ejecutando == 'abortar')
480 				{
481 					$.post("servicios/proyecto.ejecutar.php", { tipo : 'parada' }, function (valor) 
482 					{
483 						if (valor+0 == -1)
484 						{
485 							AttentionBox.showMessage( _("Error interno solicitando la parada")+". Error: "+valor);
486 						}
487 						else
488 						{
489 							ejecutando = 'parada'; // Parada correcta
490 							autoref.iniciarServicioEjecucion (servicio);
491 						}
492 					});
493 				}
494 				else 
495 				{
496 					ventana_ejecucion.ejecutando(false);
497 					ejecutando = 'parada';
498 					ventana_ejecucion.reejecutar(tipo);
499 					
500 				}
501 			}
502 		}

643 		/**
644 			Abre un diálogo de configuración de compilación
645 		    @function abrirVentanaConfiguracion
646 		    @param {string} nombre Nombre del proyecto
647 		    @param {string[]} mispropiedades Propiedades del compilador
648 		    @param {string[]} miscompiladores Lista de compiladores
649 		    @param {int} idcompilador ID del compilador utilizado
650 		    @memberof EntornoTrabajo 
651 		*/
652 		this.abrirVentanaConfiguracion = function (nombre, mispropiedades, miscompiladores, idcompilador)
653 		{
654 			// Generales
655 			var accionCambioCompilador = function() {
656 				// Si cambia el compilador hay que cambiar todas las variables
657 				var idcompilador = $(this).val();
658 				// Cerramos el diálogo y lanzamos la ventana nueva
659 				AttentionBox.closeWindow (null, "yes", "RELOAD", idcompilador);
660 			};
661 
662 			var misinputs = [
663 						        { caption : _("Nombre"), rel:"proyecto.nombre", id: "conf_proyecto_nombre", value: nombre },
664 						        { caption : _("Compilador"), rel:"proyecto.idcompilador", id: "conf_proyecto_idcompilador", 
665 						        			type: "select", values: miscompiladores, selectvalue: idcompilador, change: accionCambioCompilador },
666 						    ];
667 
668 			// Específicas de cada proyecto
669 			if (mispropiedades != undefined)
670 			$.each(mispropiedades, function(nombre, valores) 
671 			{
672 				// Desactivamos los input que tienen ya valores por defecto
673 				var valor = valores[1];
674 				var disabled = true;
675 				if (valores[1] == "") 
676 				{
677 					disabled = false;
678 					valor = valores[2];
679 				}
680 				misinputs[misinputs.length] = { caption : nombre, rel: valores[0], id: "propiedad_proyecto_"+valores[0], value: valor, disabled: disabled };
681 			});
682 
683 			AttentionBox.showMessage( _("Propiedades del proyecto"),
684 			{
685 						    inputs : misinputs,
686 						    buttons : 
687 						    [
688 						        { caption : _("Cancelar"), cancel: true },
689 						        { caption : _("Modificar") },
690 						    ],
691 							callback: function(action, inputs)
692     						{
693     							if (action == "RELOAD")
694     							{
695     								// Lanzamos la ventana nueva
696     								autoref.configurar (inputs);
697     							}
698     							else if (action == "Modificar")
699     							{
700     								// Evitamos mandar nada por rapidez
701     								if (inputs[0].value!='')
702 									{
703 										// Cargando...
704 										cargador.activar();
705 
706 										var contenido_post = {};
707 										for (i=0; i<misinputs.length; i++)
708 										{	
709 											contenido_post[misinputs[i].rel] = inputs[i].value; 
710 										}
711 										
712 										contenido_post['accion'] = 'modificar'; 
713 
714 										// Mandamos la modificación
715 										if (action ==  _("Modificar"))
716 												$.post("servicios/proyecto.editar.php", 
717 												contenido_post, 
718 												function(valor)
719 												{
720 													if (valor+0 != 0)
721 														AttentionBox.showMessage( _("No se pudo modificar el proyecto")+". Error:"+valor);
722 													else
723 														AttentionBox.showMessage( _("Cambios realizados. Se recargará la página"), 
724 															{	
725 																buttons : 
726 							    								[ { caption : _("Recargar"), cancel: true } ],
727 																callback: function ()
728 	    														{
729 	    															document.location.reload();
730 	    														}
731 	    													});
732 													// A la vuelta recargamos el explorador
733 													cargador.desactivar();
734 												});
735 										else cargador.desactivar();
736 									}
737 								}
738 								
739 							}
740 			});
741 		}

745 		    @function configurarUsuarios
