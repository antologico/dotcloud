 11 function Editor (controlador)
 12 { 
 13 		// --------------------------------------------------------
 14 		// Creación de la ventana en el body
 15 		// --------------------------------------------------------
 16 
 17 		// $id de la ventana
 18 		$("body").append ('<div id="editor_capa"><div class="scroll-pane"><div id="solapas"></div></div><div id="editor_contenedor" class="editor"></div></div>');	
 19 		// --------------------------------------------------------
 20 		var editor 					= $("#editor_contenedor");
 21 		var contenedor				= $("#editor_capa");
 22 		var solapas 				= $("#solapas");
 23 		var controlador				= controlador;
 24 		var autoref					= this;				// Autoreferencia del objeto, útil para llamadas dentro de funciones
 25 		var mieditor				= null;				// Editor de códigos. En prinpcio no está creado, se crea bajo demanda 
 26 		// --------------------------------------------------------
 27 		// Control de solapas
 28 		var id_editor_activo		= -1;				// ID para solapa creada
 29 		var lista_editores			= new Array(); 		// Para saber a qué editor corresponde el archivo
 30 														// Contiene ( [archivo] = [id, contenido] )
 31 		var num_editores			= 0; 				// Para saber cuántos archivos editados hay
 32 		
 33 		//	-----------------------------------------------------		
 34 		//	Cerrar documento
 35 		//	-----------------------------------------------------			
 36 
 37 
 38 		/**
 39 			Cierra una pestaña del editor
 40 		    @function cerrarEditor 
 41 		    @param {string} archivo Nombre del archivo
 42 		    @memberof Editor
 43 		*/
 44 		this.cerrarEditor = function (archivo)
 45 		{
 46    			// Cerramos la solapa y el editor que no usamos
 47     		if ($('#capa_solapa_'+lista_editores[archivo][0]))
 48     		{
 49 	    		$('#capa_solapa_'+lista_editores[archivo][0]).remove();
 50 	    						// Borramos de la lista de activos
 51 				
 52 				lista_editores[archivo] = null;
 53 				id_editor_activo = -1; // Inactivos todos	
 54 				num_editores --;
 55 	    		
 56 	    		// Buscamos una capa nueva que activar
 57 	    		// Si hay más solapas abiertas, dejamos abierta la primera que se se abrió
 58 	    		if (num_editores > 0)
 59 	    		{
 60 	    			for (var abierto in lista_editores)
 61 					{
 62 						if (lista_editores[abierto] != null)
 63 						{
 64 	    					// activamos la siguiente solapa
 65 	    					this.activarEditor (abierto);
 66 	    					break;
 67 	    				}
 68 	    			}
 69 	    		}
 70 	    		else 
 71 	    		{
 72 	    			mieditor 			= null;
 73 	    			$("#editor_codigo").remove();
 74 	    			
 75 	    		}
 76 	    		solapas.width(num_editores*230+"px");
 77     		}
 78 		}
 79 
 80 
 81 		/**
 82 			Crea una pestaña para un archivo nuevo
 83 		    @function crearEditor
 84 		    @param {string} archivo Nombre del archivo
 85 		    @param {string} contenido Contenido del archivo
 86 		    @param {string} [línea] Línea que debe aparecer marcada
 87 		    @memberof Editor
 88 		*/
 89 		this.crearEditor = function (archivo, contenido, linea)
 90 		{
 91 			if (archivo != undefined)
 92 			{
 93 				// Nombre corto para la solapa
 94 				var nombre_fichero = archivo.substring(archivo.lastIndexOf("\/")+1,archivo.length);
 95 
 96 				// --------------------------------------------------------
 97 				// Actualizamos índices
 98 				var id_editor  				= ++num_editores; 	// Solo crece
 99 				lista_editores[archivo] 	= [ id_editor, contenido ];
100 				
101 				// --------------------------------------------------------
102 				// Creamos una nueva solapa al principio				
103 				solapas.prepend ('<div class="solapa" id="capa_solapa_'+id_editor+'"><input type="button" id="solapa_'+id_editor+'" 	\
104 					 class="nombre" value="'+nombre_fichero+'" rel="'+id_editor+'" ref="'+archivo+'" ><input rel="'+id_editor+'" 		\
105 					 class="cerrar" id="solapa_cerrar_'+id_editor+'" type="button"></div>');
106 				$('#capa_solapa_'+id_editor).addClass('solapa_activa');
107 				$('#capa_solapa_'+id_editor).hide();
108 				$('#capa_solapa_'+id_editor).show(200);
109 					
110 
111 				// --------------------------------------------------------
112 				$('#solapa_'+id_editor).click (function() 
113 						{ 
114 							var id_capa = $(this).attr('rel');
115 							autoref.activarEditor (archivo);
116 						});
117 
118 				$('#solapa_cerrar_'+id_editor).click (function() 
119 						{ 
120 							var id_capa = $(this).attr('rel');
121 
122 							AttentionBox.showMessage( _("Está cerrando el documento sin guardarlo"),
123 							{
124 							    buttons : 
125 							    [
126 							        { caption : _("Cancelar"), cancel: true },
127 							        { caption : _("Cerrar sin guardar") },
128 							        { caption : _("Guardar") }
129 							    ],
130 								callback: function(action, inputs)
131 	    						{
132 	    							if (action ==  _("Cancelar"))
133 	    							{
134 	    								// No se hace nada, se cierra el diálogo y listo
135 	    							}
136 	    							else if (action ==  _("Guardar"))
137 	    							{
138 	    								// Se guarda el archivo
139 	    								var micontenido = null;
140 	    								var miarchivo 	= null;
141 
142 	    								// Si el archivo no está cargado en el editor, se busca en los arrays
143 	    								if (id_editor 	!= id_editor_activo)
144 	    								{
145 	    									miarchivo = archivo;
146 	    									contenido = lista_editores[archivo][1];
147 	    								}
148 	    								var error = controlador.guardarArchivo(miarchivo, micontenido);
149 	    								// y cerramos
150 	    								if (error == 0) controlador.cerrarArchivo(archivo);
151 	    							}
152 	    							else if (action ==  _("Cerrar sin guardar"))
153 	    							{
154 	    								// Cerramos
155 	    								controlador.cerrarArchivo(archivo);
156 	    							}
157 	    						}
158 	    					});
159 
160 						});
161 
162 				// --------------------------------------------------------				
163 				// Ampliamos la capa para generar la scrollbar en caso de ser necesario
164 				solapas.width(num_editores*230+"px");
165 
166 
167 				// Creamos un nuevo editor
168 				if (num_editores == 1)
169 				{
170 					editor.append ('<pre id="editor_codigo" class="pre_editor"></pre>');
171 					mieditor = ace.edit('editor_codigo'); 	
172 				}
173 				
174 				// Aplicación del tema del editor
175 				mieditor.setTheme("ace/theme/twilight");
176 				
177 				// Cargamos el contenido
178 				this.activarEditor (archivo, linea);
179 			}
180 		}

183 		/**
184 			Muestra la pestaña ya abierta del archivo indicado
185 		    @function activarEditor 
186 		    @param {string} archivo Nombre del archivo
187 		    @param {string} [línea] Línea que debe aparecer marcada
188 		    @memberof Editor
189 		*/
190 		this.activarEditor = function (archivo, linea)
191 		{
192 			if (lista_editores[archivo] !=undefined)
193 			{
194 				id_editor = lista_editores[archivo][0]; // Buscamos el id en la lista
195 				if (id_editor != id_editor_activo)
196 				{
197 					// Desactiva la marcada y activa la nueva
198 					// desactivamos la solapa vieja
199 					if (id_editor_activo != -1)
200 					{
201 						$('#capa_solapa_'+id_editor_activo).removeClass('solapa_activa');
202 						// guardamos los valores del viejo
203 						var archivo_antiguo = $('#solapa_'+id_editor_activo).attr('ref');
204 						lista_editores[archivo_antiguo] = [ id_editor_activo, mieditor.getValue() ];
205 
206 					}
207 					$('#capa_solapa_'+id_editor).addClass('solapa_activa');
208 					var archivo_nuevo = $('#solapa_'+id_editor).attr('ref');
209 					// activamos el editor nuevo
210 					mieditor.setValue (lista_editores[archivo_nuevo][1], 0);
211 					mieditor.clearSelection();
212 	           		mieditor.getSelection().moveCursorFileStart ();
213 					
214 					// Marcamos la nueva como activa
215 					id_editor_activo = id_editor;
216 				}
217 
218 				if (linea != null) 
219 				{
220 					mieditor.getSelection().moveCursorTo (linea-1, 0);
221 					mieditor.getSelection().setSelectionAnchor (linea-1, 0);
222 					mieditor.getSelection().selectLine();
223 				} 
224 
225 				mieditor.getSession().setMode("ace/mode/"+this.conseguirModoTexto(archivo));		
226 			}
227 		}

229 		/**
230 			Devuelve el nombre del archivo que está en edición
231 		    @function archivoActivo 
232 		    @memberof Editor
233 		*/
234 		this.archivoActivo = function ()
235 		{
236 			if (id_editor_activo != -1)
237 			{
238 				return $('#solapa_'+id_editor_activo).attr('ref');
239 			}
240 			return null;
241 		}

268 		/**
269 			Devuelve el tipo de coloración de sintaxis del archivo indicado
270 		    @function conseguirModoTexto
271 		    @param {string} archivo Nombre del archivo
272 		    @memberof Editor
273 		*/
274 		this.conseguirModoTexto = function (archivo)
275 		{
276 			var extension = archivo.substring(archivo.lastIndexOf(".")+1,archivo.length);
277 
278 			if (extension == 'java') 								return 'java';
279 			else if (extension == 'html') 							return 'html';
280 			else if (extension == 'js') 							return 'javascript';
281 			else if (extension == 'jsp') 							return 'jsp';
282 			else if (extension == 'sh') 							return 'sh';
283 			else if ((extension == 'c') || (extension == 'cpp'))	return 'c_cpp';
284 			else if (extension == 'jade') 							return 'jade';
285 			else if (extension == 'perl') 							return 'perl';
286 			else if (extension == 'json') 							return 'json';
287 			else if (extension == 'php') 							return 'php';
288 			else return 'text';
289 		}

294 		/**
295 			Registra los eventos de teclado para la asignación de teclas rápidas
296 			@function detectarTeclas
297 		    @param {string} evento Evento de teclado
298 		    @memberof Editor
299 		*/
300 		this.detectarTeclas = function (event)
301 		{
302 			if (!((tecla = event.which) && event.ctrlKey) && !(event.which == 19)) return true;
303 		    
304 		    switch(tecla)
305 		    {
306 		    	case 17: // Ctrl + Q
307 		    		controlador.salir ();
308 		    		break;
309 		    	case 18: // Ctrl + R
310 		    		controlador.ejecutar('ejecucion');
311 		    		break;
312 		    	case 19: // Ctrl + S
313 		    		controlador.guardarArchivo ();
314 		    		break;
315 		    	case 20: // Ctrl + T
316 		    		controlador.ejecutar('compilacion');
317 		    		break;	
318 		    	default:
319 		    		return true;
320 		    }
321 		    return false;
322 		}
