1 /**
  2  	Componente gráfico con la ventana de ejecución
  3 	
  4 	@author Antonio Juan Sánchez Martín
  5 	@class VentanaEjecucion
  6 	@version 0.9
  7 	@param {Contenedor} contenedor - Referencia al componente padre
  8 */	
  9 
 10 function VentanaEjecucion (controlador)
 11 { 
 12 		// --------------------------------------------------------
 13 		// Creación de la ventana en el body
 14 		// --------------------------------------------------------
 15 		$("body").append ('<div id="ventana_ejecucion" class="fondo_carga"><div id="datos_ejecucion" class="dialogo"><img id="en_ejecucion" src="imagenes/en_ejecucion.gif" alt="..."/><input type="button" class="cerrar" id="boton_cerrado_ejecucion"><pre class="scroll-pane" id="panel_ejecucion_salida"></pre><input id="panel_entrada_ejecucion" /><div class="scroll-pane" id="panel_lista_salida"></div><input type="button" id="reejecutar" /></div></div>');	
 16 		var ventana 		= $("#ventana_ejecucion");									// $id de la capa de carga
 17 		var boton_cerrado	= $("#boton_cerrado_ejecucion");							// $id del botón de cerrado de la ventana
 18 		var ejecucion		= $("#en_ejecucion");										// $id de la imagen de ejecución
 19 		var boton_reejecutar= $("#reejecutar");											// $id del botón de reejecucion
 20 		var cargador		= new Cargador ("cargador_ejecucion", "datos_ejecucion");	// $id de la capa de carga
 21 		var salida_pre		= $("#panel_ejecucion_salida");								// $id del panel de escritura	
 22 		var entrada			= $("#panel_entrada_ejecucion");							// $id del panel de entrada de datos
 23 		var salida_listado	= $("#panel_lista_salida");									// $id del panel de escritura	
 24 		// --------------------------------------------------------
 25 		var estado			= 'parada';
 26 		var autoref			= this;		
 27 		var controlador		= controlador;												// Padre de la ejecución
 28 		// --------------------------------------------------------
 29 		var detectartecla   				= false;
 30 		var detectartecla_accion_anterior   = null;
 31 		// --------------------------------------------------------
 32 		// Activación del cierre
 33 		boton_cerrado.click (function()		{ controlador.pararServicioEjecucion() });
 34 		boton_reejecutar.click (function()	{ autoref.detectarTeclas (false); controlador.reejecutar($(this).attr('ref')); });
 35 		
 36 		/**
 37 			Abre la ventana de ejecución
 38 		    @function mostrar
 39 		    @memberof VentanaEjecucion
 40 		*/
 41 		this.mostrar = function ()
 42 		{
 43 			ventana.fadeIn ('slow');
 44 			cargador.activar();
 45 			// Activación del cierre
 46 			boton_cerrado.click (function()	{ autoref.ocultar() });
 47 		}
 48 
 49 
 50 		/**
 51 			Asigna a un botón la propiedad seleccionada
 52 		    @function detectarTeclas
 53 		    @param {int} detectar Activa (1) / desactiva (0) la detección de eventos de teclado
 54 		    @memberof VentanaEjecucion
 55 		*/
 56 		this.detectarTeclas = function (detectar)
 57 		{
 58 			if (detectar)
 59 			{
 60 				
 61 				if (detectartecla == false)
 62 				{
 63 					controlador.enviarDatoEjecucion ("");
 64 
 65 					entrada.keypress(function (evento) { 
 66 							
 67 								evento.preventDefault(); 
 68 								// Asignamos char
 69 								var tecla = evento.which;
 70 								if (tecla == 13) tecla = "\n";
 71 								else tecla = String.fromCharCode(evento.which);
 72 								// Lo sacamos por pantalla
 73 								autoref.imprimir(tecla);
 74 								controlador.enviarDatoEjecucion (tecla);	
 75 							
 76 						});
 77 
 78 					detectartecla = true;
 79 				}
 80 			}
 81 			else
 82 			{
 83 				if (detectartecla == true)
 84 				{
 85 					entrada.keypress (autoref.detectartecla_accion_anterior);
 86 					detectartecla = false;
 87 				}
 88 			}
 89 		}
 90 
 91 		/**
 92 			Activa o desactiva la capa de carga
 93 		    @function ejecutando
 94 		    @param {int} en_ejecucion En ejecución (1) / parado (0)
 95 		    @memberof VentanaEjecucion
 96 		*/
 97 		this.ejecutando = function (en_ejecucion)
 98 		{
 99 			if (en_ejecucion == true)
100 			{
101 				ejecucion.show ();
102 				boton_reejecutar.hide ();
103 			}
104 			else
105 			{
106 				ejecucion.hide ();
107 				boton_reejecutar.show ();
108 			}
109 		}
110 
111 		/**
112 			Asigna la propiedad de reejecuión al botón de acción
113 		    @function reejecutar
114 		    @param {string} tipo Tipo de acción
115 		    @memberof VentanaEjecucion
116 		*/
117 		this.reejecutar = function (tipo)
118 		{
119 			boton_reejecutar.removeClass ();
120 			boton_reejecutar.attr ("ref", tipo);
121 			boton_reejecutar.addClass (tipo);
122 			boton_reejecutar.show ();
123 		}
124 
125 		/**
126 			Imprime los resultados en la salida tipo consola
127 		    @function imprimir
128 		    @param {string[]} resultados Tipo de acción
129 		    @memberof VentanaEjecucion
130 		*/
131 		this.imprimir = function (resultados)
132 		{
133 			this.detectarTeclas (true);
134 
135 
136 			salida_pre.show();
137 			salida_listado.hide();
138 			salida_pre.append (resultados);
139 		}
140 
141 		/**
142 			Lista y formatea los resultados de la compilación
143 		    @function listar
144 		    @param {string[]} resultados Tipo de acción
145 		    @memberof VentanaEjecucion
146 		*/
147 		this.listar = function (resultados)
148 		{
149 			this.detectarTeclas (false);
150 
151 
152 			salida_pre.hide();
153 			salida_listado.show();
154 
155 			var resultado = "<ul>";
156 
157 			for(var json in resultados)
158 				for(var errores in resultados[json])
159 					for(var linea in resultados[json][errores])
160 					{
161 						var error = 	resultados[json][errores][linea]['error'];
162 						var numlinea = 	resultados[json][errores][linea]['linea'];
163 						var archivo = 	resultados[json][errores][linea]['archivo'];
164 						var mensaje = 	resultados[json][errores][linea]['mensaje'];
165 						
166 						if (error != undefined)
167 						if (error != "")	
168 						{
169 							resultado 		+=  '<li';
170 							var marcado 	= 0;
171 							if (mensaje != undefined)
172 							{
173 								marcado 	= 	1;
174 								resultado 	+=  " class='marcado'><a";
175 								resultado 	+=  " linea='"+numlinea+"' ";
176 								resultado 	+=  " archivo='"+archivo+"' ";
177 								resultado 	+=  " mensaje='"+mensaje+"' ";
178 							}
179 							
180 							resultado += '>'+resultados[json][errores][linea]['error']
181 							if (marcado) resultado += '</a>';
182 							resultado += '</li>';
183 						}
184 					}
185 			resultado += "</ul>";
186 
187 			salida_listado.append (resultado);
188 
189 			// Añadimos una acción a todos los ítems marcados
190 			var lis = salida_listado.find("li.marcado");
191 			if (lis.length>0)
192 			{
193 				for(var i=0; i<lis.length; i++)
194 				{
195 					// Abrirá el documento que tenga señalado
196 					$(lis).find("a").click ( function ()	{
197 							autoref.ocultar ();
198 							controlador.cargarArchivo ("/"+$(this).attr('archivo'), $(this).attr('linea'));
199 						});
200 
201 					resultado += "+";
202 				}
203 			}
204 		}
205 
206 		/**
207 			Limpia las consolas de salida
208 		    @function limpiar
209 		    @memberof VentanaEjecucion
210 		*/
211 		this.limpiar = function ()
212 		{
213 			// Limpiamos la salida
214 			salida_pre.html ("");
215 			salida_listado.html ("");
216 		}
217 
218 
219 		/**
220 			Oculta la ventana de ejecución
221 		    @function ocultar
222 		    @memberof VentanaEjecucion
223 		*/
224 		this.ocultar = function ()
225 		{
226 			// Antes de ocultar, se manda parar los procesos en ejecución 
227 			ventana.fadeOut('slow');
228 			this.detectarTeclas (false);
229 		}
230 }
